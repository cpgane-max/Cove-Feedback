🔧 Google Setup Guide

One-time setup to connect this page to your Google Sheet, Drive folder, and enable editing. Takes about 10 minutes.

Step 1 — Create a Google Drive folder for media

  1. Open Google Drive → New → Folder
  2. Name it e.g. Cove Building Issues – Media
  3. Open the folder and copy the ID from the URL:
    drive.google.com/drive/folders/FOLDER_ID_HERE

Step 2 — Replace the Apps Script with the updated version

Go to your Sheet → Extensions → Apps Script, delete everything, and paste this. Replace YOUR_DRIVE_FOLDER_ID with your folder ID from Step 1.

const SHEET_ID  = "1OYjgiJQxFW67iJZDkypSIR9lrZTlk0h3s9N1Tc5w508";
const FOLDER_ID = "YOUR_DRIVE_FOLDER_ID";
const TZ        = "Australia/Sydney";

function sydneyTimestamp() {
  return Utilities.formatDate(new Date(), TZ, "dd/MM/yyyy HH:mm:ss");
}

function doGet(e) {
  var ss    = SpreadsheetApp.openById(SHEET_ID);
  var sheet = ss.getSheets()[0];
  var rows  = sheet.getDataRange().getValues();
  var headers = rows[0];
  var data = rows.slice(1).map(function(row, i) {
    var obj = {rowIndex: i + 2};
    headers.forEach(function(h, j){ obj[h] = row[j]; });
    return obj;
  });
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var data   = JSON.parse(e.postData.contents);
  var ss     = SpreadsheetApp.openById(SHEET_ID);
  var sheet  = ss.getSheets()[0];
  var folder = DriveApp.getFolderById(FOLDER_ID);

  // Upload files to Drive
  var mediaLinks = [];
  if (data.files && data.files.length > 0) {
    data.files.forEach(function(f) {
      var bytes = Utilities.base64Decode(f.data);
      var blob  = Utilities.newBlob(bytes, f.mimeType, f.name);
      var file  = folder.createFile(blob);
      file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      mediaLinks.push(file.getUrl());
    });
  }

  if (data.action === "update" && data.rowIndex) {
    // UPDATE existing row
    var row = parseInt(data.rowIndex);
    var headers = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0];
    var existingMedia = sheet.getRange(row, headers.indexOf("Media")+1).getValue();
    var allMedia = [existingMedia, mediaLinks.join(", ")].filter(Boolean).join(", ");
    sheet.getRange(row, 1, 1, 11).setValues([[
      data["Date/Time"], data["Resident Name"], data["How Logged"],
      data["Issue"], data["Action Taken"], data["Response to Resident"],
      data["Cost"], allMedia,
      data["Other Comments"], data["Last Updated"], data["Logged By"]
    ]]);
  } else {
    // NEW row
    sheet.appendRow([
      data["Date/Time"], data["Resident Name"], data["How Logged"],
      data["Issue"], data["Action Taken"], data["Response to Resident"],
      data["Cost"], mediaLinks.join(", "),
      data["Other Comments"], data["Last Updated"] || sydneyTimestamp(), data["Logged By"]
    ]);
  }

  return ContentService
    .createTextOutput(JSON.stringify({result:"success"}))
    .setMimeType(ContentService.MimeType.JSON);
}

Step 3 — Deploy / Re-deploy as a Web App

  1. Save the script, then click Deploy → New deployment
  2. Type: Web app · Execute as: Me · Access: Anyone
  3. Click Deploy and authorize — copy the new /exec URL
  4. The URL is already hardcoded in this page — if it changed, update SCRIPT_URL in the HTML
⚠️ The doGet function (which powers the calendar and issue list) requires the same Web App URL. Make sure you re-deploy after replacing the script — the calendar won't load on the old deployment.

Step 4 — Host on GitHub Pages

  1. Upload index.html to your GitHub repo
  2. Settings → Pages → source: main branch
